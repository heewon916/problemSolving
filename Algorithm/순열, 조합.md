def permutation(arr, r):
    arr = sorted(arr)
    used = [0] * len(arr)
  
    def generate(chosen):
        if len(chosen) == r:
            print(chosen)
            return
        for i in range(len(arr)):
            if not used[i]:
                used[i] = 1
                chosen.append(arr[i])
                generate(chosen)
                chosen.pop()
                used[i] = 0
  
    generate([])
```

def combination(arr, r):
    arr.sort()

    def generate(chosen):
        if len(chosen) == r:
            print(chosen)
            return
        start = arr.index(chosen[-1]) + 1 if chosen else 0
        for i in range(start, len(arr)):
            chosen.append(arr[i])
            generate(chosen)
            chosen.pop()

    generate([])
```
